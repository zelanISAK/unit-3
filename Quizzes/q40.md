```pycon
from kivymd.app import MDApp
import warnings
warnings.simplefilter('ignore')

class q40demo(MDApp):
    def build(self):
        return

    def change_color(self):
        if self.root.ids.main1.md_bg_color == "white":
            self.root.ids.main1.md_bg_color = "black"
            self.root.ids.btn1.md_bg_color = "red"
            self.root.ids.txt.text_color = "white"
        else:
            self.root.ids.main1.md_bg_color = "white"
            self.root.ids.btn1.md_bg_color = "blue"
            self.root.ids.txt.text_color = "black"


test = q40demo()
test.run()
```
